rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // For development/testing - more permissive rules
    match /images/{clientId}/{fileName} {
      // Allow read for all requests
      allow read: if true;
      
      // Allow write with file size and type validation
      allow write: if request.resource.size < 10 * 1024 * 1024 &&
                      request.resource.contentType.matches('image/.*');
      
      // Allow delete for all requests
      allow delete: if true;
    }
    
    // General images directory with same permissions
    match /images/{allPaths=**} {
      allow read: if true;
      allow write: if request.resource.size < 10 * 1024 * 1024 &&
                      request.resource.contentType.matches('image/.*');
      allow delete: if true;
    }
  }
}
